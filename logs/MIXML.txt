<?xml version="1.0" encoding="UTF-8"?>
<configuration>

    <!--Donde me dejará los archivos -->
    <property name="LOGS" value="./logs" />

    <!--Donde me dejará los mensajes de consola -->
    <appender name="Console"
              class="ch.qos.logback.core.ConsoleAppender">
        <layout class="ch.qos.logback.classic.PatternLayout">
            <Pattern>
                %black(%d{ISO8601}) %highlight(%-5level) [%blue(%t)] %yellow(%C{1.}): %msg%n%throwable
            </Pattern>
        </layout>
    </appender>


    <!-- >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>> A R C H I V O  1  P A R A  E R R O R  L E V E L  <<<<<<<<<<<<<<<<<<<<<< -->

    <appender name="nivelError"
              class="ch.qos.logback.core.rolling.RollingFileAppender">
        <file>${LOGS}/nivelError.log</file>
        <encoder
                class="ch.qos.logback.classic.encoder.PatternLayoutEncoder">
            <Pattern>%d %p %C{1.} [%t] %m%n</Pattern>
        </encoder>


        <rollingPolicy
                class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <!-- rollover daily and when the file reaches 1 MegaBytes -->
            <fileNamePattern>${LOGS}/archived/nivelError-%d{yyyy-MM-dd}.%i.log
            </fileNamePattern>
            <timeBasedFileNamingAndTriggeringPolicy
                    class="ch.qos.logback.core.rolling.SizeAndTimeBasedFNATP">
                <maxFileSize>1MB</maxFileSize>
            </timeBasedFileNamingAndTriggeringPolicy>
        </rollingPolicy>
    </appender>



    <!-- >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>A R C H I V O  2  P A R A  D E B U G  L E V E L  <<<<<<<<<<<<<<<<<<<<<<< -->

    <appender name="nivelDebug"
              class="ch.qos.logback.core.rolling.RollingFileAppender">
        <file>${LOGS}/nivelDebug.log</file>
        <encoder
                class="ch.qos.logback.classic.encoder.PatternLayoutEncoder">
            <Pattern>%d %p %C{1.} [%t] %m%n</Pattern>
        </encoder>


        <rollingPolicy
                class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <!-- rollover daily and when the file reaches 1 MegaBytes -->
            <fileNamePattern>${LOGS}/archived/nivelDebug-%d{yyyy-MM-dd}.%i.log
            </fileNamePattern>
            <timeBasedFileNamingAndTriggeringPolicy
                    class="ch.qos.logback.core.rolling.SizeAndTimeBasedFNATP">
                <maxFileSize>1MB</maxFileSize>
            </timeBasedFileNamingAndTriggeringPolicy>
        </rollingPolicy>
    </appender>






    <!-- LOG para los ERROR level en mi archivo de errores -->
    <root level="ERROR">
        <appender-ref ref="nivelError" />
    </root>


    <!-- LOG para los DEBUG level  -->
    <root level="INFO">
        <appender-ref ref="nivelDebug" />
    </root>



</configuration>
